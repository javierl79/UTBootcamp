\f0\fs24 \cf0 Sub Stock_Ticker()\
 \
  'test spreadsheet loop\
  \
    Dim ws As Worksheet\
    Dim starting_ws As Worksheet\
    Set starting_ws = ActiveSheet\
\
    For Each ws In ThisWorkbook.Worksheets\
    ws.Activate\
    Columns("J").NumberFormat = "#,###"\
\
' Up to here\
     \
      ' Variable for Stock Volume\
  Dim Stock_Ticker As String\
\
  ' Variable for Stock Total\
  Dim Stock_Total As Double\
  Stock_Total = 0\
\
  ' Variable for each Stock\
  Dim Summary_Table_Row As Integer\
  Summary_Table_Row = 2\
  \
  ' Variable to find last Row\
  Dim Row As Long\
     \
  ' Returns the row count from the bottom up on a spreadsheet\
            Row = Cells(Rows.Count, 1).End(xlUp).Row\
\
  ' Loop through all stocks\
            For i = 2 To Row\
\
\
    ' Checks to see if the second row of the first column still has the same value as the previsous value of the same first column\
            If Cells(i + 1, 1).Value <> Cells(i, 1).Value Then\
\
      ' Set the Stock Ticker\
            Stock_Ticker = Cells(i, 1).Value\
\
      ' Sum volume located in the seventh column\
            Stock_Total = Stock_Total + Cells(i, 7).Value\
\
      ' Print the Ticker in the Summary Table\
            Range("I" & Summary_Table_Row).Value = Stock_Ticker\
\
      ' Print the Total Amount to the Summary Table\
            Range("J" & Summary_Table_Row).Value = Stock_Total\
\
      ' Add one to the summary table row\
            Summary_Table_Row = Summary_Table_Row + 1\
      \
      ' Reset the Stock Total\
            Stock_Total = 0\
\
    ' If the cell immediately following a row is the same brand...\
            Else\
\
      ' Add to the Stock Total\
            Stock_Total = Stock_Total + Cells(i, 7).Value\
\
            End If\
    \
    Next i\
   \
Next\
   'test end part of code\
   \
  \
starting_ws.Activate\
End Sub\
\
